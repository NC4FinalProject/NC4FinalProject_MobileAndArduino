//블루투스
#include <SoftwareSerial.h>

//디스플레이
#include <Wire.h>
#include <LiquidCrystal_I2C.h>

// 블루투스 모듈 포트 설정 (HM10)
#define BT_RXD 3
#define BT_TXD 2
SoftwareSerial BTSerial(BT_RXD, BT_TXD);

LiquidCrystal_I2C lcd(0x27,16,2);

void setup() {
  // 블루투스
  Serial.begin(9600);  // PC에서 모니터링하기 위한 시리얼 통신 시작
  BTSerial.begin(9600);
  Serial.println("Bluetooth initial");

  // 디스플레이
  lcd.init();  // LCD초기 설정
  lcd.backlight(); // LCD초기 설정
  lcd.setCursor(0,0); //텍스트가 LCD에 나타날 위치
  lcd.print("Lecture: JAVA");
  lcd.setCursor(0,1);
  lcd.print("Progress: 25\%");
}

void loop() {

  char c;
  if (Serial.available()) {
    c = Serial.read();
    BTSerial.print(c);
  }
  if (BTSerial.available()) {
    c = BTSerial.read();
    Serial.print(c);
  }
}
